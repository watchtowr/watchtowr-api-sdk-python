## get_list_service_listing

List all discovered service assets, ordered by last seen date.

### Example Usage

```python
import watchtowr_complete_watchtowr_platform_api_documentation_python
from watchtowr_complete_watchtowr_platform_api_documentation_python import CompleteWatchtowrPlatformAPIDocumentation
from watchtowr_complete_watchtowr_platform_api_documentation_python.utils import parse_datetime


with CompleteWatchtowrPlatformAPIDocumentation(
    bearer="<YOUR_BEARER_TOKEN_HERE>",
) as complete_watchtowr_platform_api_documentation:

    res = complete_watchtowr_platform_api_documentation.service_discovery.get_list_service_listing(page=1, page_size=10, created_from=parse_datetime("2022-02-22 22:00:00"), created_to=parse_datetime("2022-02-23 22:00:00"), updated_from=parse_datetime("2022-02-22 22:00:00"), updated_to=parse_datetime("2022-02-23 22:00:00"), search="1.2.3.4", countries="US,UK", technology="react", ports="22/TCP,443/TCP,3389/UDP", port_numbers="80,443", port_types="TCP,UDP", port_services="SSH,HTTP", service_type_ids="1,2,3", business_unit_ids="1,2,3", sort_by=watchtowr_complete_watchtowr_platform_api_documentation_python.SortBy.LAST_SEEN, order_by=watchtowr_complete_watchtowr_platform_api_documentation_python.OrderBy.DESC, suppression_filter=watchtowr_complete_watchtowr_platform_api_documentation_python.SuppressionFilter.NON_SUPPRESSED)

    # Handle response
    print(res)

```
